[docs]def get_post_str():
    sdtdate = datetime.datetime.now()
    year = sdtdate.year
    hm = sdtdate.strftime('%H%M')
    sdtdate = sdtdate.timetuple()
    jdate = sdtdate.tm_yday
    post_str = '{}{:03}{}'.format(year, jdate, hm)
    return post_str
